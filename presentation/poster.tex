\begin{block}{Motivation and Target Estimands}
\justifying
Ordinal responses encode rankings, but not distances between categories. Assigning scores and estimating a mean difference therefore adds an untestable measurement assumption.

\vspace{0.45em}
\keybox{Target causal effects on the observed response distribution, \\ not coefficients on an arbitrary latent scale.}

\vspace{0.5em}
For response category $j=1,\ldots,J$, define
\[
\pi_j(d)=\Pr\{Y_i(d)=j\},
\qquad
\Delta_j=\pi_j(1)-\pi_j(0).
\]
Cumulative effects are
\[
\Delta_{\ge j}=\Pr\{Y_i(1)\ge j\}-\Pr\{Y_i(0)\ge j\}.
\]
These quantities are invariant to admissible transformations of the latent scale.
\end{block}

\begin{block}{Single-Index Representation}
\justifying
Observed categories arise from a threshold-crossing model with latent outcome $Y_i^*$:
\[
Y_i=j \iff \alpha_{j-1}<Y_i^*\le \alpha_j,
\qquad
Y_i^*=\tau D_i+h(X_i)+\varepsilon_i.
\]
The coefficient $\tau$ shifts a latent index, but its magnitude depends on location and scale normalizations. It is not itself an observed-scale causal effect.

\begin{block}{Why Cardinalization Does Not Identify the Effect}
\justifying
Let $c(1)<\cdots<c(J)$ be any numerical coding of the categories. The corresponding mean effect is
\[
\tau_c
=\mathbb E[c(Y_i(1))-c(Y_i(0))]
=\sum_{j=1}^J c(j)\Delta_j.
\]
Because ordinal data do not identify the spacing in $c(\cdot)$, neither the magnitude nor, in general, the sign of $\tau_c$ is ordinally identified.

\vspace{0.35em}
\assumptionbox{Cardinalization result}{Different monotone codings can imply different causal conclusions while preserving exactly the same observed ordering.}
\end{block}

\begin{block}{Toy Example: The Sign Flips}
Suppose treatment redistributes probability across three ordered categories:
\[
(\Delta_1,\Delta_2,\Delta_3)=(-0.15,\;0.25,\;-0.10).
\]
Then two admissible codings yield opposite signs:
\[
\begin{aligned}
 c_A=(1,2,3):\quad &\tau_{c_A}= 1 \cdot (-0.15) + 2 \cdot 0.25 + 3 \cdot (-0.10) =\mathbf{+0.05},\\
 c_B=(1,2,6):\quad &\tau_{c_B}= 1 \cdot (-0.15) + 2 \cdot 0.25 + 6 \cdot (-0.10) =\mathbf{-0.25}.
\end{aligned}
\]

\begin{block}{Identification of Distributional Effects}
Under standard identification conditions, the full distribution of potential ordinal outcomes is identified.

\vspace{0.3em}
\textbf{Assumptions (SUTVA, Conditional Ignorability, and Positivity):}
\[
Y_i(d) \perp D_i \mid X_i \qquad \text{and} \qquad 0 < \Pr(D_i = d \mid X_i) < 1 \quad \forall \, d \in \{0, 1\}.
\]

We identify the marginal potential category probabilities via standardization:
\[
\pi_j(d)=\mathbb E_X\!\left[\Pr(Y_i=j\mid D_i=d,X_i)\right].
\]

\begin{itemize}
  \item For randomized experiments, random assignment supplies ignorability. Thus the causal target is identified.
  \item The remaining challenge is estimating the conditional ordinal response distribution.
\end{itemize}
\end{block}

\begin{block}{Estimation Problem}
For a latent index $m(d,x)$ and conditional error CDF $F_{\varepsilon\mid D,X}$,
\[
\begin{aligned}
p_j(d,x)
=&\;F_{\varepsilon\mid D,X}\!\left(\alpha_j-m(d,x)\mid d,x\right)\\
&-F_{\varepsilon\mid D,X}\!\left(\alpha_{j-1}-m(d,x)\mid d,x\right).
\end{aligned}
\]

\begin{itemize}
  \item If the normal/logistic error family and index specification are correct, ordered probit/logit are efficient and adequate.
  \item However, we cannot \textit{ex ante} sure whether they are correct.
\end{itemize}
\end{block}

\begin{block}{Two Threats to Estimation}
\vspace{0.5em}
\begin{columns}[t,totalwidth=\linewidth]
\begin{column}{0.485\linewidth}
\threatbox{1. Error distribution}{Normal or logistic errors cannot represent skewness, multimodality, polarization, or treatment-dependent dispersion.}
\end{column}
\begin{column}{0.485\linewidth}
\threatbox{2. Single-index structure}{$Y_i^*=\tau D_i+h(X_i)+\varepsilon_i$ may fail when treatment changes shape rather than only location.}
\end{column}
\end{columns}
\vspace{0.5em}
\end{block}

\begin{block}{Conditional Normalizing Flows}
Let $Z_i\sim N(0,1)$ and transform it using an invertible conditional map:
\[
\varepsilon_i=T_\phi(Z_i\mid D_i,X_i),
\qquad
Z_i=T_\phi^{-1}(\varepsilon_i\mid D_i,X_i).
\]
The implied CDF is available as:
\[
F_{\varepsilon\mid D,X}(e\mid d,x)
=\Phi\!\left(T_\phi^{-1}(e\mid d,x)\right).
\]

  \vspace{1cm}

\begin{itemize}
  \item Conditional Normalizing Flow models guarantee more flexible likelihood-based distribution estimation.
  \item Monotone rational-quadratic splines (Durkan et al., 2019) retain invertibility and permit analytical CDF evaluation.
\end{itemize}
\end{block}

\begin{block}{Structured and Model-Free Specifications}
\renewcommand{\arraystretch}{1.25}
\begin{center}
\begin{tabular}{>{\bfseries}p{0.35\linewidth}p{0.29\linewidth}p{0.31\linewidth}}
\toprule
 & Structured flow & Model-free flow \\
\midrule
Systematic component & $\tau d+h_\eta(x)$ & learned nonparametrically \\
Latent distribution & conditional flow & conditional flow \\
Relaxes & error family & error family and index \\
Strength & efficiency & robustness \\
Cost & residual index risk & greater data demand \\
\bottomrule
\end{tabular}
\end{center}
Both specifications target the same standardized category-specific and cumulative effects.
\end{block}

\begin{block}{Likelihood and Consistency}
\justifying
The parameters $\Theta = \{\tau, h, \phi, \alpha\}$ maximize the sample log-likelihood:
\[
\ell_n(\Theta) = \frac{1}{n}\sum_{i=1}^n \sum_{j=1}^J \mathbf{1}(Y_i = j) \log p_j(D_i, X_i; \Theta).
\]

\textbf{Assumption C.3 (Flow Approximation \& Identification):}
The flow family $\{F_\phi\}$ is rich enough to approximate the true error CDF $F_0$ arbitrarily well. The parameters are identified up to a positive affine transformation $(a+c\alpha, c\beta, c\tau)$ with $c > 0$.

\vspace{0.3em}
\textbf{Assumption C.4 (Regularity Conditions):}
The parameter space is compact, the likelihood satisfies standard continuity and differentiability conditions, and the model is correctly specified.

\vspace{0.5em}
\assumptionbox{Theorem C.2 (Consistency of the Flow Estimator)}{
Suppose the latent outcome model and threshold structure are correctly specified, and Assumptions C.3--C.4 hold. Then, as $n \to \infty$, the maximum likelihood estimator is consistent up to a positive affine transformation:
\[
\| \widehat{\Theta} - \Theta_0 \| = o_p(1).
\]
}
\end{block}

\begin{block}{Monte Carlo Simulation}

\begin{itemize}\setlength\itemsep{0.25em}
  \item \textbf{Polarization:} a bimodal latent error distribution.
  \item \textbf{Index failure:} nonlinear treatment-response structure.
  \item Estimators: empirical frequencies, ordered probit, ordered logit, structured flow, and model-free flow
\end{itemize}

\begin{block}{Application: Authoritarian Propaganda}
\justifying
Replicating Mattingly et al.(2025)'s survey experiment on how Chinese state media affects public opinion on relative preference for the United States versus China.

\vspace{0.3em}
\begin{itemize}\setlength\itemsep{0.25em}
  \item \textbf{Parametric Models (Probit/Logit)} Overestimating the conversion of citizens into extreme supporters (\emph{Strongly Prefer China}).
  \item \textbf{Flexible Flow Estimators} show the propaganda's primary impact is concentrated in \textbf{intermediate categories}.
  \item \textbf{Political Science Implication:} Authoritarian propaganda is persuasive not by creating extreme ideologues, but by softening opposition and winning over moderate/indifferent citizens.
\end{itemize}

\begin{block}{Conclusion}
\begin{enumerate}\setlength\itemsep{0.35em}
  \item \textbf{Cardinalized mean effects are not identified from the ordinal outcomes.}
  \item \textbf{Category-specific and cumulative probability effects are causally interpretable under standard identification assumptions.}
  \item \textbf{Conditional flows separate substantive conclusions from restrictive error-distribution and index assumptions.}
\end{enumerate}
\vspace{0.25em}
\end{block}
